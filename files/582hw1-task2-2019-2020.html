dataframe1['B365diff']=B365diff

dataframe1['PSdiff']=PSdiff



mergeallbookmakersdifference=pd.concat([dataframe1['B365diff'],dataframe1['BWdiff'],dataframe1['IWdiff'],dataframe1['PSdiff']])
mergebookmakerdraw=pd.concat([dataframe1['B365D_p_drawnormalized'],dataframe1['BWD_p_drawnormalized'],dataframe1['IWD_p_drawnormalized'],dataframe1['PSD_p_drawnormalized']])
mergeactualdraw=pd.concat([dataframe1['FTR'],dataframe1['FTR'],dataframe1['FTR'],dataframe1['FTR']])
mergeddataframe1=pd.DataFrame({'bookmakerdiff':mergeallbookmakersdifference,'bookmakerdraw':mergebookmakerdraw,'actualdraw':mergeactualdraw})

px=mergeddataframe1['bookmakerdiff']
py=mergeddataframe1['bookmakerdraw']
plt.scatter(px,py)

y, bins=np.histogram(mergeddataframe1['bookmakerdiff'],bins = [-1,-0.8,-0.6,-0.4,-0.2,0,0.2,0.4,0.6,0.8,1])
x=np.histogram(mergeddataframe1[mergeddataframe1.actualdraw=='D']['bookmakerdiff'],bins)[0]

for j in range(len(y)):   #JUST TO PREVENT ERROR OF 0/0 SITUATION I EMBED THIS FUNCTION
    if y[j]==0:
        y[j]=1
        
rng=[]
rng.append((bins[:-1] +bins[1:] )/2)
plt.scatter(rng,x/y)
plt.xlabel('homewin-awaywin')
plt.ylabel('draw')
plt.title('4 bookmakers together')
